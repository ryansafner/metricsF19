# Our data comes from wage1 in the wooldridge package

wages<-wooldridge::wage1
ggplot(data = wages)+
  aes(x = exper,
      y = wage)+
  geom_point()+
  geom_smooth(method="lm")+
  scale_y_continuous(labels=scales::dollar)+

library(broom)
gender_reg<-lm(wage~exper+female, data = wages) %>%
  tidy()
